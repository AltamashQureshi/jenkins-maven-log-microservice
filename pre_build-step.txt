#!/bin/bash
set -e

APP_NAME="log-generator"
VERSION="1.0.0"

echo "🚀 Creating project structure..."

mkdir -p ${APP_NAME}/src/main/java/com/example
cd ${APP_NAME}

echo "📄 Creating Java log generator..."

cat << 'EOF' > src/main/java/com/example/LogGenerator.java
package com.example;

import java.io.FileWriter;
import java.io.PrintWriter;
import java.time.LocalDateTime;
import java.util.Random;
import java.util.UUID;

public class LogGenerator {

    private static final Random random = new Random();
    private static final String SERVICE_NAME = "log-generator-service";
    private static final String ENV =
            System.getenv().getOrDefault("APP_ENV", "dev");
    private static final String LOG_FILE =
            System.getenv().getOrDefault("LOG_FILE", "app.log");

    public static void main(String[] args) throws Exception {
        while (true) {
            String requestId = UUID.randomUUID().toString();
            long latency = random.nextInt(900) + 100;
            int logType = random.nextInt(10);

            if (logType <= 5) {
                writeLog("INFO","Request processed successfully",requestId,latency,null);
            } else if (logType <= 7) {
                writeLog("WARN","High response time detected",requestId,latency,"threshold=800ms");
            } else {
                writeLog("ERROR","Unhandled exception occurred",requestId,null,"Database timeout");
            }

            Thread.sleep(1000);
        }
    }

    private static void writeLog(String level, String message,
                                 String requestId, Long latency, String error)
            throws Exception {

        String log = String.format(
            "{ \"timestamp\": \"%s\", \"level\": \"%s\", \"service\": \"%s\", \"env\": \"%s\", " +
            "\"requestId\": \"%s\", \"message\": \"%s\", \"latency_ms\": %s, \"error\": %s }",
            LocalDateTime.now(), level, SERVICE_NAME, ENV,
            requestId, message,
            latency == null ? "null" : latency,
            error == null ? "null" : "\"" + error + "\""
        );

        try (PrintWriter out = new PrintWriter(new FileWriter(LOG_FILE, true))) {
            out.println(log);
        }

        System.out.println(log);
    }
}
EOF

echo "📄 Creating pom.xml..."

cat << EOF > pom.xml
<project xmlns="http://maven.apache.org/POM/4.0.0">
  <modelVersion>4.0.0</modelVersion>
  <groupId>com.example</groupId>
  <artifactId>${APP_NAME}</artifactId>
  <version>${VERSION}</version>

  <properties>
    <maven.compiler.source>17</maven.compiler.source>
    <maven.compiler.target>17</maven.compiler.target>
  </properties>

  <build>
    <plugins>
      <plugin>
        <groupId>org.apache.maven.plugins</groupId>
        <artifactId>maven-jar-plugin</artifactId>
        <version>3.3.0</version>
        <configuration>
          <archive>
            <manifest>
              <mainClass>com.example.LogGenerator</mainClass>
            </manifest>
          </archive>
        </configuration>
      </plugin>
    </plugins>
  </build>
</project>
EOF

echo "🔨 Building JAR..."
mvn clean package

echo "✅ Build complete"
echo "📦 Artifact: target/${APP_NAME}-${VERSION}.jar"
